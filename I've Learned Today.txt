-- INNER JOIN , Mutuals lang nila yung ipapakita sa isat isa, mutual ng ON
SELECT *
FROM employees e
INNER JOIN departments d
    ON e.id = d.id;
    
-- LEFT JOIN , i j lahat ng LEFT (employees) sa Departments, tas i a output lng ng depts is yung may match sa LEFT join which is employees
SELECT *
FROM employees e
LEFT JOIN departments d
    ON e.id = d.id;

-- RIGHT JOIN , i j joing lang is yung RIGHT (depts) kaya pinakita lahat ng dept yung id na meron sila, tas ipapakita lang nmn ng employees yung may match.
SELECT *
FROM employees e
RIGHT JOIN departments d
    ON e.id = d.id;

-- FULL JOIN, Literall na lahat papakita kahit unmatches
SELECT *
FROM employees e
FULL JOIN departments d
    ON e.id = d.id;
